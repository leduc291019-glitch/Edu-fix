             this.maSach= maSach;
             this.tenSach=tenSach;
             this.giaSach=giaSach;
             }
       #định nghĩa phương thức trườu tượng
       public double thanhTien(){
              return this.thanhTien;
             }

              }

public class SachGiaoKhoa extends Sach {
       private int soLuong;
       public SachGiaoKhoa(String maSach,String tenSach,double giaSach,int soLuong){
       super(maSach,tenSach,giaSach)
       this.soLuong=soLuong;
       }
       @override
       public void thanhTien(){
             return this.giaSach * this.soLuong;
       }
       public int getsoLuong(){
             return soLuong;
       }
	public void setSoLuong(int soLuong){
		return this.soLuong;
	}
}
